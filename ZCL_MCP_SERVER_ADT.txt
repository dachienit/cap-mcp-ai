CLASS zcl_cap_adt_handler DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  PUBLIC SECTION.
    INTERFACES if_http_extension.

  PRIVATE SECTION.
    TYPES: BEGIN OF ty_request,
             objecturl  TYPE string,
             sourceurl  TYPE string,
             sourcecode TYPE string,
           END OF ty_request.

    TYPES: BEGIN OF ty_response,
             success TYPE abap_bool,
             message TYPE string,
           END OF ty_response.
           
    METHODS extract_lock_handle
      IMPORTING iv_xml TYPE string
      RETURNING VALUE(rv_handle) TYPE string.
ENDCLASS.

CLASS zcl_cap_adt_handler IMPLEMENTATION.

  METHOD if_http_extension~handle_request.
    DATA: ls_request  TYPE ty_request,
          ls_response TYPE ty_response.
          
    DATA: lo_client TYPE REF TO if_http_client,
          lv_host   TYPE string,
          lv_port   TYPE string,
          lv_url    TYPE string.

    DATA: lv_csrf_token TYPE string,
          lv_conn_id    TYPE string,
          lv_lockhandle TYPE string,
          lv_status     TYPE i.

    " Set response format as JSON
    server->response->set_header_field( name = 'Content-Type' value = 'application/json' ).

    " Only allow POST method
    IF server->request->get_method( ) <> 'POST'.
      server->response->set_status( code = 405 reason = 'Method Not Allowed' ).
      ls_response-success = abap_false.
      ls_response-message = 'Only POST is supported'.
      server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).
      RETURN.
    ENDIF.

    " Parse inbound JSON utilizing standard /UI2 class (Available on NetWeaver 7.40+)
    DATA(lv_body) = server->request->get_cdata( ).
    /ui2/cl_json=>deserialize( EXPORTING json = lv_body CHANGING data = ls_request ).

    IF ls_request-objecturl IS INITIAL OR ls_request-sourceurl IS INITIAL.
      server->response->set_status( code = 400 reason = 'Bad Request' ).
      ls_response-success = abap_false.
      ls_response-message = 'Missing objectUrl or sourceUrl in JSON body'.
      server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).
      RETURN.
    ENDIF.

    " Generate connection ID
    lv_conn_id = cl_system_uuid=>create_uuid_c32_static( ).
    lv_host = server->request->get_header_field( '~server_name' ).
    lv_port = server->request->get_header_field( '~server_port' ).
    
    " Pass logon cookies from incoming request to internal ADT request 
    " (MYSAPSSO2 or SAP_SESSIONID to preserve Principal Propagation)
    DATA(lt_inc_cookies) = server->request->get_cookies( ).

    " =======================================
    " STEP 1: Fetch CSRF Token
    " =======================================
    lv_url = |http://{ lv_host }:{ lv_port }/sap/bc/adt/core/discovery|.
    cl_http_client=>create_by_url( EXPORTING url = lv_url IMPORTING client = lo_client ).
    
    LOOP AT lt_inc_cookies INTO DATA(ls_cook).
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.

    lo_client->propertytype_logon_popup = if_http_client=>co_disabled.
    lo_client->request->set_method( 'GET' ).
    lo_client->request->set_header_field( name = 'X-CSRF-Token' value = 'Fetch' ).
    lo_client->request->set_header_field( name = 'X-sap-adt-session-type' value = 'stateful' ).
    lo_client->request->set_header_field( name = 'sap-adt-connection-id' value = lv_conn_id ).
    
    lo_client->send( EXCEPTIONS OTHERS = 1 ).
    lo_client->receive( EXCEPTIONS OTHERS = 1 ).
    
    lv_csrf_token = lo_client->response->get_header_field( 'X-CSRF-Token' ).
    DATA(lt_state_cookies) = lo_client->response->get_cookies( ).
    lo_client->close( ).

    IF lv_csrf_token IS INITIAL.
      ls_response-success = abap_false.
      ls_response-message = 'Step 1: Failed to fetch CSRF.' .
      server->response->set_status( code = 500 reason = 'Internal Error' ).
      server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).
      RETURN.
    ENDIF.

    " =======================================
    " STEP 2: Lock Object
    " =======================================
    lv_url = |http://{ lv_host }:{ lv_port }{ ls_request-objecturl }?_action=LOCK&accessMode=MODIFY|.
    cl_http_client=>create_by_url( EXPORTING url = lv_url IMPORTING client = lo_client ).

    LOOP AT lt_inc_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.
    LOOP AT lt_state_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.

    lo_client->request->set_method( 'POST' ).
    lo_client->request->set_header_field( name = 'X-CSRF-Token' value = lv_csrf_token ).
    lo_client->request->set_header_field( name = 'X-sap-adt-session-type' value = 'stateful' ).
    lo_client->request->set_header_field( name = 'sap-adt-connection-id' value = lv_conn_id ).
    
    lo_client->send( EXCEPTIONS OTHERS = 1 ).
    lo_client->receive( EXCEPTIONS OTHERS = 1 ).
    
    lv_status = lo_client->response->get_status( ).
    IF lv_status >= 400.
      ls_response-success = abap_false.
      ls_response-message = |Step 2: Lock failed (HTTP { lv_status })|.
      server->response->set_status( code = 500 reason = 'Internal Error' ).
      server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).
      lo_client->close( ).
      RETURN.
    ENDIF.

    lv_lockhandle = lo_client->response->get_header_field( 'X-sap-adt-lock-handle' ).
    IF lv_lockhandle IS INITIAL.
      lv_lockhandle = lo_client->response->get_header_field( 'X-sap-adt-lockhandle' ).
    ENDIF.
    
    DATA(lv_xml) = lo_client->response->get_cdata( ).
    IF lv_lockhandle IS INITIAL.
       lv_lockhandle = extract_lock_handle( lv_xml ).
    ENDIF.

    DATA(lt_lock_cookies) = lo_client->response->get_cookies( ).
    LOOP AT lt_lock_cookies INTO ls_cook.
       READ TABLE lt_state_cookies WITH KEY name = ls_cook-name ASSIGNING FIELD-SYMBOL(<fs_ck>).
       IF sy-subrc = 0.
          <fs_ck> = ls_cook.
       ELSE.
          APPEND ls_cook TO lt_state_cookies.
       ENDIF.
    ENDLOOP.
    lo_client->close( ).

    IF lv_lockhandle IS INITIAL.
      ls_response-success = abap_false.
      ls_response-message = 'Step 2: Code was locked but couldn''t extract Handle XML.'.
      server->response->set_status( code = 500 reason = 'Internal Error' ).
      server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).
      RETURN.
    ENDIF.

    " =======================================
    " STEP 3: Set Source
    " =======================================
    lv_url = |http://{ lv_host }:{ lv_port }{ ls_request-sourceurl }?lockHandle={ cl_http_utility=>escape_url( lv_lockhandle ) }|.
    cl_http_client=>create_by_url( EXPORTING url = lv_url IMPORTING client = lo_client ).

    LOOP AT lt_inc_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.
    LOOP AT lt_state_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.

    lo_client->request->set_method( 'PUT' ).
    lo_client->request->set_header_field( name = 'X-CSRF-Token' value = lv_csrf_token ).
    lo_client->request->set_header_field( name = 'X-sap-adt-session-type' value = 'stateful' ).
    lo_client->request->set_header_field( name = 'sap-adt-connection-id' value = lv_conn_id ).
    lo_client->request->set_header_field( name = 'Content-Type' value = 'text/plain; charset=utf-8' ).
    
    lo_client->request->set_cdata( ls_request-sourcecode ).
    
    lo_client->send( EXCEPTIONS OTHERS = 1 ).
    lo_client->receive( EXCEPTIONS OTHERS = 1 ).
    
    lv_status = lo_client->response->get_status( ).
    DATA(lv_put_response) = lo_client->response->get_cdata( ).
    lo_client->close( ).
    
    IF lv_status >= 400.
      ls_response-success = abap_false.
      ls_response-message = |Step 3: Save failed (HTTP { lv_status }) - { lv_put_response }|.
      " Skip to unlock logic below
    ELSE.
      ls_response-success = abap_true.
      ls_response-message = 'Source code saved successfully!'.
    ENDIF.

    " =======================================
    " STEP 4: Unlock
    " =======================================
    lv_url = |http://{ lv_host }:{ lv_port }{ ls_request-objecturl }?_action=UNLOCK&lockHandle={ cl_http_utility=>escape_url( lv_lockhandle ) }|.
    cl_http_client=>create_by_url( EXPORTING url = lv_url IMPORTING client = lo_client ).

    LOOP AT lt_inc_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.
    LOOP AT lt_state_cookies INTO ls_cook.
      lo_client->request->set_cookie( name = ls_cook-name value = ls_cook-value ).
    ENDLOOP.

    lo_client->request->set_method( 'POST' ). 
    lo_client->request->set_header_field( name = 'X-CSRF-Token' value = lv_csrf_token ).
    lo_client->request->set_header_field( name = 'X-sap-adt-session-type' value = 'stateful' ).
    lo_client->request->set_header_field( name = 'sap-adt-connection-id' value = lv_conn_id ).
    
    lo_client->send( EXCEPTIONS OTHERS = 1 ).
    lo_client->receive( EXCEPTIONS OTHERS = 1 ).
    lo_client->close( ).

    " Output Response
    IF ls_response-success = abap_true.
      server->response->set_status( code = 200 reason = 'OK' ).
    ELSE.
      server->response->set_status( code = 500 reason = 'Internal Server Error' ).
    ENDIF.
    server->response->set_cdata( /ui2/cl_json=>serialize( data = ls_response ) ).

  ENDMETHOD.
  
  METHOD extract_lock_handle.
    FIND REGEX '<[a-zA-Z0-9:]*lockHandle[^>]*>([^<]+)</' IN iv_xml SUBMATCHES rv_handle.
    IF rv_handle IS INITIAL AND iv_xml NS '<'.
      rv_handle = iv_xml.
    ENDIF.
  ENDMETHOD.

ENDCLASS.
